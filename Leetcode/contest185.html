    1418. Display Table of Food Orders in a Restaurant

    It is easyunderstanding, but it is hard to choose data stucture. I will use the fastest code with my explaination. 

    class Solution {
public:
  vector<vector<string>> displayTable(vector<vector<string>>& orders) {
    map<int, unordered_map<string, int>> tally;
    set<string> allFoods;
    for (auto& o : orders) {
      int tab = stoi(o[1]);
      string food = o[2];
      allFoods.insert(food);
      tally[tab][food]++;
    }
    /*We have a map, so we can know each table and the food,number they ordered.
    tally:
    |table number(int)| food
    3                 ["water", 3], ["chicken" 4]...
    */
    vector<vector<string>> ans;
    ans.push_back({"Table"});
    for (auto f : allFoods)
      ans.back().push_back(f);

    for (auto t : tally) {
      string tab = to_string(t.first);
      ans.push_back({tab});
      for (auto f : allFoods) {
        //if the food table #tab ordered, push_back the number
        if (t.second.count(f))
          ans.back().push_back(to_string((t.second)[f]));
        else
          ans.back().push_back("0");
      }
    }
    return ans;
  }
};
